import { Meta, Canvas, Story } from '@storybook/blocks';
import { html } from 'lit';

<Meta title="Atoms/Tooltip" />

export const Template = ({ content, margin = '10rem', ...args }) => {
  const getTooltip = (evt) => evt.srcElement.parentElement.querySelector('.pixel-tooltip');
  const onMouseover = (evt) => {
    getTooltip(evt).classList.add('pixel-tooltip--show');
  };
  const onMouseout = (evt) => {
    getTooltip(evt).classList.remove('pixel-tooltip--show');
  };
  return html`<div class="pixel-tooltip-container" style="margin-${args.position.split('-')[0]}: ${margin};">
    <div class="pixel-tooltip pixel-tooltip--${args.position}">
      <div class="pixel-tooltip__arrow pixel-tooltip__arrow--${args.position}"></div>
      <div class="pixel-tooltip__wrapper">tooltip content</div>
    </div>
    <button class="pixel-button" @mouseover=${onMouseover} @mouseout=${onMouseout}>${content} Tooltip</button>
  </div>`;
};
